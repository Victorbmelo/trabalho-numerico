float solucao(int k, int cont)
{
    float solucoes[size];

    if (k==0)
    {
        // for (b=0; b<size; b++)
        return gs[cont];
    }
    else if (cont==size-1)
    {
        solucoes[cont] = solucao(k-1, cont) + (matriz[size-1][size])/(matriz[size-1][size-1]);
        return solucoes[cont];
    }
    else
    {
        solucoes[cont]= solucao(k-1, cont) + 1/matriz[cont-1][cont-1]*(matriz[cont-1][size]-regressao(cont));
        return solucoes[cont];
    }
}
